Backend Entry Point (app.py) Analysis

1. 비동기 환경 설정 (Monkey Patching)
from gevent import monkey
monkey.patch_all()  # 영상을 계속 분석하면서 동시에 웹 요청도 받아야함 (비동기로 한 번에 여러 일을 막힘없이 처리)

[설명] 동시성 처리를 위한 기초 공사

파이썬의 표준 라이브러리를 비동기 방식으로 동작하게 만드는 Monkey Patch입니다. AI 영상 분석과 웹 소켓 통신이 동시에 일어나는 관제 서버의 특성상, 어느 한쪽이 멈추지 않도록 만드는 가장 중요한 첫 줄입니다.

2~5. 서버 라이브러리 및 DB/SocketIO 초기화
import os
import warnings
import atexit 
from dotenv import load_dotenv
from flask import Flask, request, Response, jsonify
from flask_socketio import SocketIO, emit
from flask_cors import CORS
from flask_migrate import Migrate 

from detectors.manager import detector_manager 
from models import db 

load_dotenv()
warnings.filterwarnings("ignore", category=FutureWarning)

app = Flask(__name__)
CORS(app)

DB_USER = os.getenv("DB_USER")
DB_PW = os.getenv("DB_PASSWORD")
DB_HOST = os.getenv("DB_HOST")
DB_PORT = os.getenv("DB_PORT")
DB_NAME = os.getenv("DB_NAME")
app.config['SQLALCHEMY_DATABASE_URI'] = f'mysql+pymysql://{DB_USER}:{DB_PW}@{DB_HOST}:{DB_PORT}/{DB_NAME}'
app.config['SQLALCHEMY_TRACK_MODIFICATIONS'] = False

db.init_app(app)
migrate = Migrate(app, db)

socketio = SocketIO(app, cors_allowed_origins="*", async_mode='gevent')
app.extensions['socketio'] = socketio

[설명] 인프라 및 통신 계층 설정

환경 변수를 로드하고 DB(SQLAlchemy)와 실시간 통신(SocketIO) 엔진을 초기화합니다. 특히 SocketIO는 async_mode='gevent' 설정을 통해 앞서 설정한 비동기 패치와 호흡을 맞춥니다.

# 서버 종료 시 실행될 함수 등록 (분석 스레드 안전 종료)
def shutdown_detectors():
    print("🛑 [System] 서버 종료 감지: 모든 분석 스레드를 정지합니다...")
    detector_manager.stop_all()

atexit.register(shutdown_detectors)

[설명] 유지보수성과 자원 관리

기능별로 Blueprint를 나누어 등록함으로써 대규모 프로젝트에서도 코드가 꼬이지 않도록 관리합니다. 또한 atexit을 통해 서버 종료 시 실행 중인 AI 분석기들을 안전하게 정지시켜 '좀비 프로세스' 발생을 방지합니다.

@socketio.on('resolve_emergency')
def handle_resolve(data):
    print(f"📡 조치 신호 전파: {data.get('alertId')}")
    emit('emergency_resolved', data, broadcast=True)

@app.route('/')

    save_path = os.path.join(app.root_path, "static", "captures")
    if not os.path.exists(save_path):

[설명] 서버 가동 및 환경 준비

서버가 시작되기 전, 분석 결과 이미지를 저장할 물리적 폴더를 스스로 점검하고 생성합니다. socketio.run을 통해 일반적인 HTTP 요청과 실시간 이벤트를 모두 처리할 수 있는 상태로 서버를 기동합니다.
